def calculate_particle_entropy(particle, temperature=None):
    """Calculate entropy based on particle properties"""
    
    # Spin contribution
    S_spin = log(2 * particle.spin + 1)
    
    # Mass ratio contribution  
    S_mass = log(1 + particle.mass / electron_mass) / 10
    
    # Charge contribution
    S_charge = 1.0 if particle.charge != 0 else 0.0
    
    # Thermal contribution (if temperature provided)
    S_thermal = 3/2 * log(T/300) if temperature else 0.0
    
    return S_spin + S_mass + S_charge + S_thermal
